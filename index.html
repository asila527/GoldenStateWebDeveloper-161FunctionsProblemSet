function isEven(num) { if (num % 2 === 0) { return 'true'; } else { return 'false'; } }

function factorial(num) { if (num === 0) { return 1; } else { for (let i = num - 1; i >= 1; i--) { num = num * i; } return num; } }

function kebabToSnake(str) { return str.replace(/[-_]/g, "_"); }
